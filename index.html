/** * VICE Unit — SA:MP <-> Supabase Bridge * ────────────────────────────────────────────────────────────── * Sits between your VICE members' MoonLoader script and your * Supabase project. The Lua script never talks to Supabase * directly — it talks to this bridge over plain HTTP, and the * bridge forwards requests to Supabase using the SERVICE ROLE key * (never give that key to any game client). * * Run with: node server.js * Requires: npm install express @supabase/supabase-js dotenv * * Environment variables (put these in a .env file next to this * file — never commit the .env file or the service role key): * SUPABASE_URL=https://busjaqntreeikycmoztw.supabase.co * SUPABASE_SERVICE_KEY= API> * BRIDGE_TOKEN= * PORT=3000 * * BRIDGE_TOKEN is a shared secret between this server and every * officer's MoonLoader script. Anyone who has it can write * seizures/arrests/status as any officer, so keep the bridge off * the public internet if you can, and treat the token like a * password. * * EVERY ENDPOINT IS GET-ONLY. This is deliberate: MoonLoader's * built-in downloadUrlToFile() — the only HTTP function guaranteed * to exist with zero extra libraries on every officer's PC — can * only do GET requests to a URL and save the response to a file. * It cannot send a POST body or custom headers. So every operation * (including writes like submitting a seizure) is a GET request * with all data encoded into the URL's query string. * * SECURITY NOTE: because of the above, the bridge token has to be * passed as a query parameter (?token=...) instead of a header. * That means it can end up in this server's own request logs and * potentially in any reverse-proxy logs in front of it. Treat it * the same as you would a password that occasionally appears in * logs: rotate it if you ever suspect it's leaked, and don't run * this bridge behind a public/shared logging proxy you don't trust. */ require('dotenv').config(); const express = require('express'); const rateLimit = require('express-rate-limit'); const { createClient } = require('@supabase/supabase-js'); const { SUPABASE_URL, SUPABASE_SERVICE_KEY, BRIDGE_TOKEN, PORT = 3000, } = process.env; if (!SUPABASE_URL || !SUPABASE_SERVICE_KEY || !BRIDGE_TOKEN) { console.error('Missing required env vars. Check SUPABASE_URL, SUPABASE_SERVICE_KEY, BRIDGE_TOKEN.'); process.exit(1); } const supabase = createClient(SUPABASE_URL, SUPABASE_SERVICE_KEY, { auth: { autoRefreshToken: false, persistSession: false }, }); const app = express(); app.use(express.json()); // ────────────────────────────────────────────────────────────── // RATE LIMITING // Separate limits for read vs write endpoints so heavy reads // don't eat into the write budget and vice versa. // ────────────────────────────────────────────────────────────── const readLimit = rateLimit({ windowMs: 60 * 1000, // 1 minute max: 60, // 60 read requests per IP per minute standardHeaders: true, legacyHeaders: false, message: { ok: false, error: 'too many requests — slow down' }, }); const writeLimit = rateLimit({ windowMs: 60 * 1000, // 1 minute max: 10, // 10 write requests per IP per minute standardHeaders: true, legacyHeaders: false, message: { ok: false, error: 'too many write requests — slow down' }, }); // ────────────────────────────────────────────────────────────── // HELPER: look up a verified officer by badge. // Used by every write endpoint. Rejects unverified/unknown badges // so a pending officer (or someone guessing badge numbers) cannot // submit logs even if they know the bridge token. // ────────────────────────────────────────────────────────────── async function requireVerifiedOfficer(badge, res) { if (!badge) { res.status(400).json({ ok: false, error: 'officer_badge required' }); return null; } const { data: officer, error } = await supabase .from('officers') .select('id, name, badge, role, verified') .eq('badge', badge.toString().trim()) .single(); if (error || !officer) { res.status(404).json({ ok: false, error: 'officer not found' }); return null; } if (!officer.verified) { res.status(403).json({ ok: false, error: 'officer account is pending verification' }); return null; } return officer; } // ────────────────────────────────────────────────────────────── // HELPER: truncate string fields to prevent database bloat from // oversized payloads. // ────────────────────────────────────────────────────────────── const trunc = (val, max = 500) => val ? val.toString().slice(0, max) : null; app.use(express.urlencoded({ extended: true })); // ────────────────────────────────────────────────────────────── // Auth middleware — accepts the token either as a header // (x-bridge-token, for any future non-MoonLoader client) or as a // query parameter (?token=..., for MoonLoader's GET-only downloadUrlToFile). // ────────────────────────────────────────────────────────────── app.use((req, res, next) => { const token = req.headers['x-bridge-token'] || req.query.token; if (token !== BRIDGE_TOKEN) { return res.status(401).json({ ok: false, error: 'unauthorized' }); } next(); }); // Simple console logger so you can see what's coming in. Strips the // token out of logged URLs so it doesn't sit in plain text in your // server's own log files. app.use((req, res, next) => { const safeUrl = req.originalUrl.replace(/([?&]token=)[^&]+/, '$1***'); console.log(`[${new Date().toISOString()}] ${req.method} ${safeUrl}`); next(); }); // ────────────────────────────────────────────────────────────── // GET /officer/lookup?badge=1234 // Looks up an officer by badge number so the Lua script can // confirm a logged-in player is a registered, verified officer // before letting them use /vice or /r. // ────────────────────────────────────────────────────────────── app.get('/officer/lookup', readLimit, async (req, res) => { const badge = (req.query.badge || '').toString().trim(); if (!badge) return res.status(400).json({ ok: false, error: 'badge required' }); const { data, error } = await supabase .from('officers') .select('id, name, badge, rank, role, verified') .eq('badge', badge) .single(); if (error) return res.status(404).json({ ok: false, error: 'officer not found' }); res.json({ ok: true, officer: data }); }); // ────────────────────────────────────────────────────────────── // GET /seizure/log // Query params: // token, officer_badge, log_type, drug, qty, unit, firearm, // firearm_qty, value, location, op_type, notes // arrestees = "Name|Charges|Evidence;;Name2|Charges2|Evidence2" // (each person is name|charges|evidence, people separated by ;; // — this is built by the Lua script, see vice_unit.lua's // buildArresteesParam()) // // GET, not POST, because MoonLoader's downloadUrlToFile (the only // HTTP function guaranteed available without extra libraries) can // only do GET requests with everything in the URL. // ────────────────────────────────────────────────────────────── app.get('/seizure/log', writeLimit, async (req, res) => { const q = req.query; if (!q.location) return res.status(400).json({ ok: false, error: 'location required' }); const officer = await requireVerifiedOfficer(q.officer_badge, res); if (!officer) return; const { data: newLog, error } = await supabase .from('seizure_logs') .insert({ officer_id: officer.id, officer_name: officer.name, officer_badge: officer.badge, log_type: trunc(q.log_type, 50), drug: trunc(q.drug, 100), drug_detail: trunc(q.drug_detail, 200), firearm: trunc(q.firearm, 100), firearm_detail: trunc(q.firearm_detail, 200), qty: q.qty ? Number(q.qty) : null, unit: trunc(q.unit || 'grams', 20), firearm_qty: q.firearm_qty ? Number(q.firearm_qty) : null, value: q.value ? Number(q.value) : 0, arrests: 0, location: trunc(q.location, 200), op_type: trunc(q.op_type || 'Other', 50), notes: trunc(q.notes, 1000), }) .select('id') .single(); if (error) return res.status(500).json({ ok: false, error: error.message }); // Parse the delimited arrestees string, if present. let arresteeRows = []; if (q.arrestees) { arresteeRows = q.arrestees .split(';;') .map(part => part.split('|')) .filter(parts => parts.length >= 1 && parts[0]) .map(([name, charges, evidence, dnaFlag, fpFlag]) => ({ log_id: newLog.id, name: (name || '').trim(), charges: (charges || '').trim(), evidence: (evidence || '').trim() || '-', dna: dnaFlag === '1', fingerprints: fpFlag === '1', })) .filter(a => a.name); } if (arresteeRows.length) { // For each arrestee, check if a suspect record exists by name (case-insensitive). // If found, attach their suspect_id so the mugshot auto-displays on the website // without the officer needing to do anything extra. If not found, the row is still // inserted — a mugshot can be added later via the website's log modal. const enriched = await Promise.all(arresteeRows.map(async row => { try { const { data: suspect } = await supabase .from('suspects') .select('id') .ilike('name', row.name) .maybeSingle(); if (suspect?.id) return { ...row, suspect_id: suspect.id }; } catch (e) { // suspects table may not exist yet — skip silently } return row; })); const { error: arrErr } = await supabase.from('log_arrestees').insert(enriched); if (arrErr) { console.error('log_arrestees insert error:', arrErr.message); } else { await supabase.from('seizure_logs').update({ arrests: enriched.length }).eq('id', newLog.id); } } res.json({ ok: true, log_id: newLog.id, arrests: arresteeRows.length }); }); // ────────────────────────────────────────────────────────────── // GET /seizure/recent?badge=1234&limit=5 // Used by /vice "My Arrests" / "Recent Arrests" / "Seizure Logs" // menu options. // ────────────────────────────────────────────────────────────── app.get('/seizure/recent', readLimit, async (req, res) => { const limit = Math.min(parseInt(req.query.limit, 10) || 10, 50); const badge = (req.query.badge || '').toString().trim(); let q = supabase .from('seizure_logs') .select('id, log_type, drug, firearm, qty, unit, value, arrests, location, officer_name, officer_badge, created_at') .order('created_at', { ascending: false }) .limit(limit); if (badge) q = q.eq('officer_badge', badge); const { data, error } = await q; if (error) return res.status(500).json({ ok: false, error: error.message }); res.json({ ok: true, logs: data }); }); // ────────────────────────────────────────────────────────────── // GET /seizure/:id // Full details for one seizure, including arrestees — used by // "Seizure Details" / "View Details" in the /vice menu. // ────────────────────────────────────────────────────────────── app.get('/seizure/:id', readLimit, async (req, res) => { const { data: log, error } = await supabase .from('seizure_logs') .select('*') .eq('id', req.params.id) .single(); if (error || !log) return res.status(404).json({ ok: false, error: 'not found' }); const { data: arrestees } = await supabase .from('log_arrestees') .select('name, evidence, charges') .eq('log_id', log.id); res.json({ ok: true, log, arrestees: arrestees || [] }); }); // ────────────────────────────────────────────────────────────── // GET /seizure/search?q=ganton&limit=10 // Backs the /vice "Search Seizures" option — matches against // location, drug, firearm, or officer name. // ────────────────────────────────────────────────────────────── app.get('/seizure/search', readLimit, async (req, res) => { const term = (req.query.q || '').toString().trim(); const limit = Math.min(parseInt(req.query.limit, 10) || 10, 50); if (!term) return res.status(400).json({ ok: false, error: 'q required' }); const { data, error } = await supabase .from('seizure_logs') .select('id, log_type, drug, firearm, qty, unit, value, location, officer_name, created_at') .or(`location.ilike.%${term}%,drug.ilike.%${term}%,firearm.ilike.%${term}%,officer_name.ilike.%${term}%`) .order('created_at', { ascending: false }) .limit(limit); if (error) return res.status(500).json({ ok: false, error: error.message }); res.json({ ok: true, logs: data }); }); // ────────────────────────────────────────────────────────────── // GET /stats?badge=1234 // Backs "Unit Statistics" in the /vice menu. // ────────────────────────────────────────────────────────────── app.get('/stats', readLimit, async (req, res) => { const badge = (req.query.badge || '').toString().trim(); const { data: mine } = badge ? await supabase.from('seizure_logs').select('value, arrests, qty, firearm_qty').eq('officer_badge', badge) : { data: [] }; const { data: all } = await supabase.from('seizure_logs').select('value, arrests, qty, firearm_qty'); const sum = (rows, key) => (rows || []).reduce((s, r) => s + (r[key] || 0), 0); res.json({ ok: true, mine: { seizures: (mine || []).length, value: sum(mine, 'value'), arrests: sum(mine, 'arrests'), firearms: sum(mine, 'firearm_qty'), }, unit: { seizures: (all || []).length, value: sum(all, 'value'), arrests: sum(all, 'arrests'), firearms: sum(all, 'firearm_qty'), }, }); }); // ────────────────────────────────────────────────────────────── // GET /status/set — backs the in-game /r command // Query params: token, officer_badge, officer_name, code, label, details, location // Upserts officer_status (one row per officer = current status) // and inserts a row into officer_status_log (full history), // mirroring how dispatch_calls works for the website's manual // "Log Call" button. // ────────────────────────────────────────────────────────────── app.get('/status/set', writeLimit, async (req, res) => { const q = req.query; if (!q.code) return res.status(400).json({ ok: false, error: 'code required' }); const officer = await requireVerifiedOfficer(q.officer_badge, res); if (!officer) return; const row = { officer_id: officer.id, officer_name: officer.name, officer_badge: officer.badge, code: q.code, label: q.label || null, details: q.details || null, location: q.location || null, updated_at: new Date().toISOString(), }; const { error: upErr } = await supabase .from('officer_status') .upsert(row, { onConflict: 'officer_id' }); if (upErr) return res.status(500).json({ ok: false, error: upErr.message }); // Best-effort history log — don't fail the request if this fails const { error: logErr } = await supabase.from('officer_status_log').insert({ officer_id: officer.id, officer_name: officer.name, officer_badge: officer.badge, code: q.code, label: q.label || null, details: q.details || null, location: q.location || null, }); if (logErr) console.error('officer_status_log insert error:', logErr.message); res.json({ ok: true }); }); // ────────────────────────────────────────────────────────────── // GET /status/all — live roster, e.g. for an in-game /unitstatus // command or for debugging the bridge. // ────────────────────────────────────────────────────────────── app.get('/status/all', readLimit, async (req, res) => { const { data, error } = await supabase .from('officer_status') .select('officer_name, officer_badge, code, label, details, location, updated_at') .order('updated_at', { ascending: false }); if (error) return res.status(500).json({ ok: false, error: error.message }); res.json({ ok: true, officers: data }); }); // ────────────────────────────────────────────────────────────── // GET /dispatch/log — backs the in-game /r ten-code command // Upserts into dispatch_calls: updates the officer's existing // active row if one exists, otherwise inserts a new one. This // prevents the dispatch log filling up with one row per /r call. // ────────────────────────────────────────────────────────────── app.get('/dispatch/log', writeLimit, async (req, res) => { const q = req.query; if (!q.code) return res.status(400).json({ ok: false, error: 'code required' }); const officer = await requireVerifiedOfficer(q.officer_badge, res); if (!officer) return; // Check for an existing active row for this officer const { data: existing } = await supabase .from('dispatch_calls') .select('id') .eq('officer_id', officer.id) .neq('status', 'resolved') .order('created_at', { ascending: false }) .limit(1) .single(); const payload = { code: trunc(q.code, 20), details: trunc(q.details || q.code, 300), location: trunc(q.location || '', 200), status: ['active','resolved'].includes(q.status) ? q.status : 'active', officer_id: officer.id, officer_name: officer.name, officer_badge: officer.badge, }; let error; if (existing?.id) { // Update the existing active row in place ({ error } = await supabase .from('dispatch_calls') .update(payload) .eq('id', existing.id)); } else { // No active row — insert a fresh one ({ error } = await supabase .from('dispatch_calls') .insert(payload)); } if (error) return res.status(500).json({ ok: false, error: error.message }); res.json({ ok: true }); }); app.listen(PORT, () => { console.log(`VICE Unit bridge listening on port ${PORT}`); }); app.get('/health', (req, res) => res.json({ ok: true })); // ────────────────────────────────────────────────────────────── // DELETE AUTH USER — removes a user from Supabase Auth entirely // so they can re-register with the same email. // Requires admin token. Only callable by verified admins. // ────────────────────────────────────────────────────────────── app.get('/officer/auth/delete', writeLimit, requireVerifiedOfficer, async (req, res) => { const { uid } = req.query; if (!uid) return res.status(400).json({ ok: false, error: 'uid required' }); const { error } = await supabase.auth.admin.deleteUser(uid); if (error) return res.status(500).json({ ok: false, error: error.message }); res.json({ ok: true }); });
